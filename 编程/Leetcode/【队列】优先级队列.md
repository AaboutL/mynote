<font color=RED>题目中有第几个、前几个、第几大、第几小等字眼时，可以考虑优先队列</font>
普通的队列是一种先进先出的数据结构，元素在队列尾追加，而从队列头删除。

在优先队列中，元素被赋予优先级。当访问元素时，具有最高优先级的元素最先删除。<font color=RED>优先队列具有最高级先出 （first in, largest out）的行为特征。不再是先进先出了</font>，这里的优先级可以是元素的大小、或者自定义的优先级，其中常见的是把元素的大小当做优先级。

首先要包含头文件#include\<queue\>, 他和queue不同的就在于我们可以自定义其中数据的优先级, 让优先级高的排在队列前面, 优先出队。

优先队列具有队列的所有特性，包括队列的基本操作，只是在这基础上添加了内部的一个排序，它本质是一个堆实现的。

## 1. 定义
 定义：priority_queue<Type, Container, Functional>
Type 就是数据类型，Container 就是容器类型（Container必须是用数组实现的容器，比如vector,deque等等，但不能用 list。STL里面默认用的是vector），Functional 就是比较的方式。
当需要用自定义的数据类型时才需要传入这三个参数，使用基本数据类型时，只需要传入数据类型，默认是大顶堆（即<font color=RED>默认是降序排列</font>）。
~~~
//升序队列，小顶堆
priority_queue <int,vector<int>,greageter<int> > q;
//降序队列，大顶堆
priority_queue <int,vector<int>,less<int> >q;

//greater和less是std实现的两个仿函数（就是使一个类的使用看上去像一个函数。其实现就是类中实现一个operator()，这个类就有了类似函数的行为，就是一个仿函数类了）
~~~

## 2. 使用示例：
### 1 基础类型类型
~~~
#include <queue>
priority_queue<int> queue; // 降序排列
priority_queue<int, vector<int>, greater<int>> queue; //升序

priority_queue<pair<int, int>> queue; // 默认按first排序
~~~

### 2 自定义类型
<font color=RED> 必须重写小于号 </font>

~~~
#include <queue>

struct State{
	int val;
	bool operator < (const State& ref) const{
		return val > ref.val; // 升序排列
	}
};
~~~

## 3. 题目示例
### 3.1 【23. 合并K个升序链表】https://leetcode.cn/problems/merge-k-sorted-lists/
给你一个链表数组，每个链表都已经按升序排列。
请你将所有链表合并到一个升序链表中，返回合并后的链表。
~~~C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    struct State{
        int val;
        ListNode* pNode;
        bool operator < (const State& state) const {
            return val > state.val;
        }
    };
    

    ListNode* mergeKLists(vector<ListNode*>& lists) {
        priority_queue<State> q;
        for(auto& list : lists){
            if(list)
                q.push({list->val, list});
        }

        ListNode* dummy = new ListNode();
        ListNode* node = dummy;
        while(!q.empty()){
            ListNode* tmp = q.top().pNode;
            q.pop();
            node->next = tmp;
            node = node->next;
            if(tmp->next){
                q.push({tmp->next->val, tmp->next});
            }
        }
        
        return dummy->next;
    }
};
~~~

### 3.2 【215. 数组中的第K个最大元素】https://leetcode.cn/problems/kth-largest-element-in-an-array/
#### 解法一：优先队列
~~~C++
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        priority_queue<int, vector<int>, greater<int>> q; // 小根堆
        for(int i = 0; i < nums.size(); i++){
            // 如果队列长度小于k，则填满；
            // 当长度等于k时，如果新元素大于队头元素，则插入队列，否则就丢掉。
            if(q.size() < k || q.top() < nums[i]) 
                q.push(nums[i]);
            
            // 由于插入新元素导致队列长度大于k，则需要把队头元素去掉。
            // 这样当遍历完后，队列中剩余的元素为从第k大 到 最大的元素。
            if(q.size() > k)
                q.pop();
        }
        return q.top();
    }
};
~~~

#### 解法二：partition + 二分
partition每次划分后，得到三部分：小于pivot部分、pivot，大于pivot部分。其中pivot所在的位置就是数组排序后最终的位置。即一次partition操作，可以确定一个元素（pivot）的最终位置。
可以根据这个确定位置与k的大小来判断，下次划分是对小于pivot部分划分，还是对大于pivot部分划分。直到某次划分的确定位置等于k，此确定位置的元素就是结果。
~~~C++
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        k = nums.size() - k;
        int lo = 0, hi = nums.size() - 1;
        while(lo <= hi){
            int j = partition(nums, lo, hi);
            if (j == k)
                return nums[j];
            else if(j < k){
                lo = j + 1;
            }
            else{
                hi = j - 1;
            }
        }
        return -1;
    }

    int partition(vector<int>& nums, int lo, int hi){
        int p = nums[lo];
        int j = lo - 1; // 此处的减1
        for(int i = lo; i <= hi; i++){
            if(nums[i] <= p){ // 和此处的‘=’都是为了处理存在多个元素和pivot相等的情况
                j++;
                swap(nums, i, j);
            }
        }
        swap(nums, lo, j); // nums[lo]作为pivot一直没有实际交换过，
		//所以此时从lo+1到j都是比nums[lo]小的元素，所以需要nums[lo]和nums[j]
		//进行交换，使得j是一个分界点。
        return j;
    }
    void swap(vector<int>& nums, int i, int j){
        int tmp = nums[i];
        nums[i] = nums[j];
        nums[j] = tmp;
    }
};
~~~

快速排序：
~~~C++
void sort(vector<int>& nums, int lo, int hi){
        if(lo >= hi) //当nums本来就是从小到大，那么需要’>’来进行返回
            return;
        int p = partition(nums, lo, hi);
        sort(nums, lo, p -1);
        sort(nums, p + 1, hi);
    }
~~~

### 3.3 【264. 丑数 II】https://leetcode.cn/problems/ugly-number-ii/
#### 解法一：优先队列
~~~C++
class Solution {
public:
    int nthUglyNumber(int n) {
        priority_queue<long, vector<long>, greater<long>> q;
        long ans = 1;
        for(int i = 1; i < n; i++){
            q.push(ans * 2);
            q.push(ans * 3);
            q.push(ans * 5);
            ans = q.top();
            q.pop();
            while(!q.empty() && q.top() == ans){
                q.pop();
            }
        }
        return ans;
    }
};
~~~

#### 解法二：动态规划
dp[i]表示第i个丑数。
因为丑数指的是2、3、5的倍数，所以后面的丑数一定是前面的某个丑数* 2、* 3、* 5 得到的
~~~C++
class Solution {
public:
    int nthUglyNumber(int n) {
        vector<int> dp(n);
        dp[0] = 1;
        int p2 = 0, p3=0, p5=0;
        for(int i = 1; i < n; i++){
            dp[i] = min( min(dp[p2] * 2, dp[p3]* 3), dp[p5]*5 );
            if(dp[p2] * 2 == dp[i])
                p2++;
            if(dp[p3] * 3 == dp[i])
                p3++;
            if(dp[p5] * 5 == dp[i])
                p5++;
        }
        return dp.back();
    }
};
~~~

### 3.4 【373. 查找和最小的 K 对数字】https://leetcode.cn/problems/find-k-pairs-with-smallest-sums/
取第二个数组的第一个元素，与第一个数组的所有元素组成tuple（sum, id1, id2），放到优先队列，第一个pair是全局最小的。后面出队一个就入队一个，入队的编号根据出队的tuple确定，具体的就是tuple中的id1和id2+1。
因为nums2[id2 + 1] >= nums2[id2]，所以nums1[id1] + nums2[id2] \<=  nums1[id1] + nums2[id2+1]。
~~~C++
class Solution {
public:
    vector<vector<int>> kSmallestPairs(vector<int>& nums1, vector<int>& nums2, int k) {
        vector<vector<int>> ans;
        priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>> > q;

        int len1 = nums1.size(), len2 = nums2.size();
        for(int i = 0; i < min(k, len1); i++){
            q.push(vector<int>{nums1[i] + nums2[0], i, 0});
        }

        while(k > 0 && !q.empty()){
            auto item = q.top();
            ans.push_back({nums1[item[1]], nums2[item[2]]});
            q.pop();
            if(item[2] < len2 - 1){
                q.push({nums1[item[1]] + nums2[item[2] + 1], item[1], item[2] + 1});
            }
            k--;
        }
        return ans;
    }
};
~~~

### 3.5 【378. 有序矩阵中第 K 小的元素】https://leetcode.cn/problems/kth-smallest-element-in-a-sorted-matrix/
本题和上面的题目类似，只不过不再是两个数组了，是n个数组。
~~~C++
class Solution {
public:
    int kthSmallest(vector<vector<int>>& matrix, int k) {
        int len = matrix.size();
        priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> q;
        int ans;
        for(int i = 0; i < len; i++){
            q.push({matrix[0][i], 0, i});
        }

        while(k > 0 && !q.empty()){
            auto item = q.top();
            q.pop();
            ans = item[0];
            if(item[1] < len -1){
                q.push({matrix[item[1] + 1][item[2]], item[1] + 1, item[2]});
            }
            k--;
        }
        return ans;
    }
};
~~~
