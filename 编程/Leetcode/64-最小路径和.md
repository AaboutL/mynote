给定一个包含非负整数的m* n的网格grid，请找出一条从左上角到右下角的路径，是的路径上的数字总和为最小；要求：每次只能向下或者向右移动一步。
示例：
||||
|--|--|--|
|1|3|1|
|1|5|1|
|4|2|1|
最小路径为 1-3-1-1-1，和为7

解法一：
```
#include <iostream>
#include <vector>
#include <math.h>

class Solution{
public:
    int solver(std::vector<std::vector<int>> &grid){
        int m = grid.size();
        int n = grid[0].size();
        std::vector<std::vector<int>> dp(m, std::vector<int>(n));
        dp[0][0] = grid[0][0];
        for (int i = 1; i < m; ++i){
            dp[i][0] = dp[i - 1][0] + grid[i][0];
        }
        for (int j = 1; j < n; ++j){
            dp[0][j] = dp[0][j - 1] + grid[0][j];
        }
        for (int i = 1; i < m; ++i){
            for (int j = 1; j < n; ++j){
                dp[i][j] = std::min(dp[i-1][j], dp[i][j -1]) + grid[i][j];
            }
        }
        return dp[m - 1][n - 1];
    }
};

int main(int argc, char** argv){
    Solution slt = Solution();
    std::vector<std::vector<int>> grid{{1,3,1},{1,5,1},{4,2,1}};
    int rst = slt.solver(grid);
    std::cout << rst << std::endl;
}
```

解法二：空间优化
```
int solver(std::vector<std::vector<int>> &grid){
        int m = grid.size();
        int n = grid[0].size();
        std::vector<int> dp(n);
        for (int i = 0; i < m; ++i){
            for (int j = 0; j < n; ++j){
                if (0 == i){
                    dp[j] = (j - 1 >= 0 ? (dp[j - 1] + grid[i][j]) : grid[i][j]);
                }
                else{
                    dp[j] = grid[i][j] + (j - 1 >=0 ? std::min(dp[j - 1], dp[j]) : dp[j]);
                }
            }
        }
        return dp[n -1];
    }
```
