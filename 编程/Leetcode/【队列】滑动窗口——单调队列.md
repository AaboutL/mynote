<font color=BLUE>普通队列</font>只能按照时间顺序对数据进行操作，如先进先出，操作的只是最老时间和最新时间这两个元素。对于最老时间和最新时间之间的元素无法处理。
<font color=BLUE>优先级队列</font>：通过维护大（小）顶堆实现快速获取最大(小)值。<font color=RED>是一种按时间进，但是按照优先级出的结构</font>。这种结构的最值与时间顺序无关。无法满足普通队列按时间顺序先进先出的操作。
<font color=blue>单调队列</font>：入队时，按照需求从队尾把小于（或大于）当前值的元素出队，从而构成单调递减（或单调递增）的队列。这样，在出队时，是按照时间顺序，可以获取滑动窗口内的最大值（最小值）。
## 单调队列
<font color=RED>单调队列用于滑动窗口问题的解决。</font>
普通的滑窗问题可以凭借扩大窗口（right++）和缩小窗口（left++），通过移出窗口和移入窗口的元素来判断如何更新状态。
对于<font color=red>窗口内求最值的问题</font>，要么需要对所有元素遍历，要么就需要新的数据结构，单调队列就是一种解决方法。
<font color=BLUE>保持单调性的方法</font>：队尾元素与当前元素比较，不符合的出队。即队尾出队是为了保持单调性，队头出队是为了更新结果。
框架：
~~~C++
class MonotonicQueue{
public:
    void push(int num){
        while(!q.empty() && q.back() < num){
            q.pop_back();//这里一定要从back出队，因为front可能是最大值
        }
        q.push_back(num);
    }

    int max(){
        return q.front();
    }

    void pop(int num){
        if(num != q.front())
            return;
        q.pop_front();
    }
private:
    deque<int> q;
};
~~~

## 示例
### 1. 【239. 滑动窗口最大值】https://leetcode.cn/problems/sliding-window-maximum/
~~~C++
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        vector<int> rst;
        MonotonicQueue q;
        for(int i = 0; i < nums.size(); i++){
            if(i < k-1){
                q.push(nums[i]);
            }
            else{
                q.push(nums[i]);
                rst.push_back(q.max());
                q.pop(nums[i - (k - 1)]);
            }
        }
        return rst;
    }
};
~~~

### 2. 【862. 和至少为 K 的最短子数组】https://leetcode.cn/problems/shortest-subarray-with-sum-at-least-k/
需要用到两个方法组合解决的题目一般为困难题；
单调队列的使用方法不都是如239题那么明显：直接把元素放到队列中。如本题中，队列中的元素是数组的下标。
本题用到了**前缀和**和**单调队列**两个方法。
<font color=RED>与子数组的和有关的题目首先要考虑</font><font color=BLUE>前缀和。</font>
~~~C++
class Solution {
public:
    int shortestSubarray(vector<int>& nums, int k) {
        int n = nums.size();
        vector<long> preSum(n+1);
        for(int i = 0; i < n; i++){
            preSum[i+1] = preSum[i] + nums[i];
        }
        int rst = n+1;
        deque<int> q;
        for(int i = 0; i < n+1; i++){
            long curSum = preSum[i];
            while(!q.empty() && curSum - preSum[q.front()] >= k){
                rst = min(rst, i - q.front());
                q.pop_front();
            }
			
			// 对队列进行处理，形成单调队列
            while(!q.empty() && curSum < preSum[q.back()]){
                q.pop_back();
            }
            q.push_back(i);
        }
        return rst == n+1 ? -1 : rst;
    }
};
~~~

### 3. 【918. 环形子数组的最大和】https://leetcode.cn/problems/maximum-sum-circular-subarray/
<font color=RED>看到环形数组，就应该想到把数组扩展成2倍</font>
<font color=RED>看到子数组的和，就应该想到前缀和。</font>如果构造的前缀和数组长度=原数组的长度+1，计算区间时，是左开右闭区间 (left, right\]，在判断子数组长度时很重要。
所以本题还是前缀和+单调队列
~~~C++
class Solution {
public:
    int maxSubarraySumCircular(vector<int>& nums) {
        int len = nums.size();
        vector<int> preSum(2 * len + 1);
        for(int i = 0; i < 2 * len; i++){
            preSum[i + 1] = preSum[i] + nums[i % len];
        }

        deque<int> q;
        int ans = INT_MIN;
        q.push_back(0);
        for(int i = 1; i < 2 * len + 1; i++){
            int curSum = preSum[i];
            while(!q.empty() && curSum < preSum[q.back()]){
                q.pop_back();
            }
            while(!q.empty() && i - q.front() > len){
                q.pop_front();
            }
            ans = max(ans, curSum - preSum[q.front()]);
            q.push_back(i);
        }
        return ans;
    }
};
~~~

### 4. 【1438. 绝对差不超过限制的最长连续子数组】https://leetcode.cn/problems/longest-continuous-subarray-with-absolute-diff-less-than-or-equal-to-limit/
本题用到了两个单调队列，一个单增，一个单减
~~~C++
class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int len = nums.size();
        deque<int> qup, qdown;
        int left = 0, ans = 0;
        for(int i = 0; i < len; i++){
            int num = nums[i];
            while(!qup.empty() && num < qup.back()){
                qup.pop_back();
            }
            qup.push_back(num);
            while(!qdown.empty() && num > qdown.back()){
                qdown.pop_back();
            }
            qdown.push_back(num);

            while(!qup.empty() && !qdown.empty() && qdown.front() - qup.front() > limit){
                if(nums[left] == qup.front()){
                    qup.pop_front();
                }
                if(nums[left] == qdown.front()){
                    qdown.pop_front();
                }
                left++;
            }
            ans = max(ans, i - left + 1);
        }
        return ans;
    }
};
~~~

### 5. 【1499. 满足不等式的最大值】https://leetcode.cn/problems/max-value-of-equation/
本题需要先根据题目中的条件做一个转换：
数组的元素在x维度是递增的，所以$x_j > x_i, 所以|x_i - x_j| = x_j - x_i$，那么可以对题目的目标进行转换：
$$
求最大的 \ y_i + y_j + |x_i - x_j| \Leftrightarrow (y_j + x_j) + (y_i - x_i)
$$
这样题目就可以用单调队列解决：队列根据$y_j - x_j$单调递减，因为遍历到当前j时，需要加上以前最大的$ (y_i - x_i)$。
~~~C++
class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        int n = points.size();
        deque<int> dq;
        int ans = INT_MIN;
        for(int i = 0; i < n; i++){
            int curdiff = points[i][1] - points[i][0];
            while(!dq.empty() && points[i][0] - points[dq.front()][0] > k){
                dq.pop_front();
            }
            if(!dq.empty()){
                ans = max(ans, points[i][1] + points[i][0] + points[dq.front()][1] - points[dq.front()][0]);
            }
            while(!dq.empty() && curdiff > points[dq.back()][1] - points[dq.back()][0]){
                dq.pop_back();
            }
            dq.push_back(i);
        }
        return ans;
    }
};
~~~
