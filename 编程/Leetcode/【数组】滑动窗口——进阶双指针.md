76题代码；
~~~
string minWindow(string& s, string& t){
	unordered_map<char, int> need, window;
	int left = 0, right = 0;
	int valid=0;
	int start = 0, len = INT_MAX;
	
	for(auto c: t) need[c]++;
	
	while(right < s.size()){
		char c = s[right];
		right++;
		if(need.count(c)){
			window[c]++;
			if(window[c] == need[c])
				valid++; // 又增加了一个满足要求的字符
		}
		while(valid == need.size()){ // 所有字符都满足
			if(right - left < len){
				start = left;
				len = right - left;
			}
			
			d = s[left];
			left++;
			if(need.count(d)){
				if(window[d] == need[d])
					valid--;
				window[d]--;
			}
		}
	}
	return len == INT_MAX? "" : s.substr(start, len);
}
