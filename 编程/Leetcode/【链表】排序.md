# 1. 插入排序
~~~C++
class Solution {
public:
    ListNode* insertionSortList(ListNode* head) {
        if(!head || !head->next)
            return head;
        
        ListNode* dummy = new ListNode(0);
        dummy->next = head;
		// lastSorted 一方面是排好序部分的最后一个结点，
		// 另一方面，lastSorted的next指针是未排序部分的第一个结点。
		// cur 用于对未排序部分进行遍历，是未排序的第一个结点。
		// 所以lastSorted的next是指向cur的。
		// 这样就不需要额外的指针保存未排序部分了。
        ListNode* lastSorted = head;
        ListNode* cur = head->next;
        while(cur){
            if(lastSorted->val <= cur->val){
                lastSorted = lastSorted->next;
            }
            else{
                ListNode* p = dummy;
                while(p->next->val <= cur->val){
                    p = p->next;
                }
                lastSorted->next = cur->next;
                cur->next = p->next;
                p->next = cur;
            }
            cur = lastSorted->next;
        }
        return dummy->next;
    }
};
~~~

# 2. 归并排序
### 1. 自顶向下的归并
先对链表从上到下递归的进行划分，然后再对有序的子链表从下到上进行合并，合并的过程中进行比较。
划分的时候需要找到链表的中间结点，从中间结点划分。找中间结点可以使用快慢指针的方法。
注意，<font color=red>每个子链表都是左闭右开的$[head, tail)$</font>
~~~C++
ListNode* mergeSort(ListNode *head){
	return sort(head, nullptr);
}
ListNode* sort(ListNode *head, ListNode *tail){
	if(!head)
		return head;
	if(head->next == tail){ 
		// head和tail相连，说明在这个区间中只有一个有效结点。
		head->next = nullptr;
		return head;
	}
	ListNode *slow = head, *fast = head;
	while(fast != tail && fast->next != tail){
		slow = slow->next;
		fast = fast->next->next;
	}
	ListNode *mid = slow;
	return merge(sort(head, mid), sort(mid, tail));
}

ListNode *merge(ListNode *head1, ListNode *head2){
	ListNode *dummy = new ListNode();
	ListNode *p = dummy, *p1 = head1, *p2 = head2;
	while(p1 && p2){
		if(p1->val <= p2->val){
			p->next = p1;
			p1 = p1->next;
		}
		else{
			p->next = p2;
			p2 = p2->next;
		}
		p = p->next;
	}
	if(p1)
		p->next = p1;
	if(p2)
		p->next = p2;
	return dummy->next;
}
~~~

### 2. 自底向上的归并
首先从长度为1的子链表开始，相邻的两两归并，得到很多长度为2的有序子链表；
然后对长度为2的子链表进行归并。。。。。。
~~~C++
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if(!head)
            return head;
        int len = 0;
        ListNode *node = head;
        while(node){
            len++;
            node = node->next;
        }

        ListNode *dummy = new ListNode(0, head);
        for(int l = 1; l < len; l*=2){
            ListNode *pre = dummy, *cur = dummy->next;
            while(cur){
                ListNode *head1 = cur;
                for(int i = 1; i < l && cur->next; i++)
                    cur = cur->next;
                ListNode *head2 = cur->next; // 对链表1进行截断
                cur->next = nullptr;

                cur = head2;
                for(int i = 1; i < l && cur->next; i++)
                    cur = cur->next;
                ListNode* next = nullptr;
                if(cur){ 
                    next = cur->next; // next是下一组的开始结点
                    cur->next = nullptr; // 对链表2进行截断
                }

                ListNode *merged = merge(head1, head2);
				
                pre->next = merged;
                while(pre->next)
                    pre = pre->next; // pre相当于下一组的dummy
                cur = next; // 下一组的开始结点
            }
        }
        return dummy->next;
    }

    ListNode* merge(ListNode* head1, ListNode* head2){
        ListNode* dummy = new ListNode(0);
        ListNode* p = dummy, *p1 = head1, *p2 = head2;
        while(p1 && p2){
            if(p1->val <= p2->val){
                p->next = p1;
                p1 = p1->next;
            }
            else{
                p->next = p2;
                p2 = p2->next;
            }
            p = p->next;
        }
        if(p1)
            p->next = p1;
        else if(p2)
            p->next = p2;
        return dummy->next;
    }
};
~~~
