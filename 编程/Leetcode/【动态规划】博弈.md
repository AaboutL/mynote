博弈问题是在一个总价值固定的游戏中，所有玩家依次选择对自己最有利的策略，最后谁获得的价值最大，谁胜出。所有玩家都选择一次为一轮，一般要求每轮玩家的先后顺序是相同的。
题目中一般是两个选手参加Alice和Bob，且固定每次都是Alice先开始，最后判断谁获胜。
因为每次选择，无论Alice还是Bob都是选择对自己最有利的情况，所以在解题时，不需要关注每一种情况是Alice还是Bob的选择，只需要记住<font color=RED>每次选择都选最优的情况</font>

## 示例
### 1. 【1406. 石子游戏 III】https://leetcode.cn/problems/stone-game-iii/
#### 解法一：超时
在这个解法中，定义 dp\[i] 为选择到第 i 堆石头时，能够得到的最大价值。这个最大价值是通过比较 选择第 i 堆、选择第 i 和 i+1 堆、选择第 i、i+1、i+2 堆时的最大值得到的。这个解法中会重复计算很多相同的情况（画出决策方案就能看出），所以最终超时了。
~~~C++
class Solution {
public:
    vector<int> dp;

    int dfs(vector<int>& stoneValue, int sum, int i){
        if(i >= stoneValue.size())
            return 0;
        for(int j = 0; j < 3 && i + j < stoneValue.size(); j++){
            int v = 0;
            for(int k = 0; k <= j; k++){
                v += stoneValue[i + k];
            }
            dp[i] = max(dp[i], sum - dfs(stoneValue, sum - v, i + j + 1));
        }
        return dp[i];
    }

    string stoneGameIII(vector<int>& stoneValue) {
        int len = stoneValue.size();
        // dp[i]表示从第i堆石头开始，分别拿1/2/3堆能取得的最大值。
        dp.resize(len, INT_MIN);
        int sum = 0; // 剩余堆的总价值
        for(auto& v : stoneValue){
            sum += v;
        }
        dfs(stoneValue, sum, 0);
        if(sum - dp[0] == dp[0])
            return "Tie";
        else if(sum - dp[0] < dp[0])
            return "Alice";
        else
            return "Bob";
    }
};
~~~

#### 解法二：记忆化搜索
所谓记忆化搜索，就是用一个状态数组记录已经计算过的状态，当下次遍历到这个状态时，就不用重新计算了，直接使用状态数组中记录的值。由此可以避免重复计算。
这里面的一个重点就是<font color=RED>相同的状态，不论计算多少次，其值是不变的</font>
~~~C++
class Solution {
public:
    vector<vector<int>> dp;

    int dfs(vector<int>& stoneValue, int sum, int i){
        if(i >= stoneValue.size())
            return 0;
        
        int maxv = INT_MIN;
        for(int j = 0; j < 3 && i + j < stoneValue.size(); j++){
            int v = 0;
            for(int k = 0; k <= j; k++){
                v += stoneValue[i + k];
            }
            if(dp[i][j] == INT_MIN)
                dp[i][j] = sum - dfs(stoneValue, sum - v, i + j + 1);
            maxv = max(maxv, dp[i][j]);
        }
        return maxv;
    }

    string stoneGameIII(vector<int>& stoneValue) {
        int len = stoneValue.size();
        // dp[i][j]表示从第i堆石头开始，分别拿1/2/3堆能取得的最大值。
        dp.resize(len, vector<int>(3, INT_MIN));
        int sum = 0; // 剩余堆的总价值
        for(auto& v : stoneValue){
            sum += v;
        }
        int maxA = dfs(stoneValue, sum, 0);
        if(sum - maxA == maxA)
            return "Tie";
        else if(sum - maxA < maxA)
            return "Alice";
        else
            return "Bob";
    }
};
~~~

#### 解法三：前缀和+记忆化搜索
主要简化了一些代码，复杂度没什么变化
~~~C++
class Solution {
public:
    vector<vector<int>> dp;
    vector<int> preSum;

    int dfs(vector<int>& stoneValue, int i){
        if(i >= stoneValue.size())
            return 0;
        
        int maxv = INT_MIN;
        for(int j = 0; j < 3 && i + j < stoneValue.size(); j++){
            if(dp[i][j] == INT_MIN)
                dp[i][j] = preSum[stoneValue.size()] - preSum[i] - dfs(stoneValue, i + j + 1);
            maxv = max(maxv, dp[i][j]);
        }
        return maxv;
    }

    string stoneGameIII(vector<int>& stoneValue) {
        int len = stoneValue.size();
        // dp[i][j]表示从第i堆石头开始，分别拿1/2/3堆能取得的最大值。
        dp.resize(len, vector<int>(3, INT_MIN));
        preSum.resize(len+1);
        for(int i = 0; i < len; i++){
            preSum[i+1] = preSum[i] + stoneValue[i];
        }
        int maxA = dfs(stoneValue, 0);
        int maxB = preSum.back() - maxA;
        if(maxB == maxA)
            return "Tie";
        else if(maxB < maxA)
            return "Alice";
        else
            return "Bob";
    }
};
~~~

#### 解法四：前缀和+动态规划
前缀和是为了方便得到从第 i 到第 len-1 堆的价值总和。

**动态规划**：博弈问题的动态规划好像都是从后往前遍历。因为是从前面的状态依赖后面的状态。
dp[i] 表示当剩余的第 i, i+1, i+2 ... len-1堆石头时，玩家能得到的最大值。
注意：不是本次选择的最大值，是当前和后面所有选择归结到当前这个状态下的最大值。当前这个状态的意思就是剩余 从 i 到len-1堆石头
~~~C++
class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int len = stoneValue.size();
        vector<int> preSum(len+1);
        for(int i = 0; i < len; i++){
            preSum[i+1] = preSum[i] + stoneValue[i];
        }

        vector<int> dp(len + 1);
        dp[len] = 0;
        for(int i = len -1; i >= 0; i--){
            int maxv = INT_MAX;
            for(int j = i+1; j <= i + 3 && j <= len; j++){
                maxv = min(maxv, dp[j]);
            }
            dp[i] = preSum[len] - preSum[i] - maxv;
        }
        int maxA = dp[0];
        int maxB = preSum[len] - maxA;
        if(maxB == maxA)
            return "Tie";
        else if(maxB < maxA)
            return "Alice";
        else
            return "Bob";
    }
};
~~~

### 2. 【877. 石子游戏】https://leetcode.cn/problems/stone-game/
#### 解法一：递归超时
~~~C++
class Solution {
public:
    vector<int> rst;
    int dfs(vector<int>& piles, int i, int sum, int l, int r){
        if(i == piles.size() - 1){
            rst[i] = sum;
            return rst[i];
        }
        rst[i] = max(sum - dfs(piles, i+1, sum - piles[l], l + 1, r), sum - dfs(piles, i+1, sum-piles[r], l, r-1));
        return rst[i];
    }

    bool stoneGame(vector<int>& piles) {
        int sum = 0;
        for(auto& n : piles){
            sum+=n;
        }
        rst.resize(piles.size());
        dfs(piles, 0, sum, 0, piles.size() - 1);
        cout << rst[0] << endl;
        if(rst[0] > sum / 2)
            return true;
        else 
            return false;
    }
};
~~~

#### 解法二：动态规划
**状态定义**：dp\[i]\[j] 表示在剩余的【i, j】区间内，先手的最佳策略与后手的最佳策略的差。
**状态转移方程**：
**当 i== j 时**
	表示只剩最后一堆，那么 dp\[i]\[j] = piles\[i];  0 <= i < piles.size();
**当i != j 时**
	$dp[i][j] = max(piles[i] - dp[i+1][j],  piles[j] - dp[i][j-1])$
推导：假设先手选择 i：先手选择能从【i, j】得到的总量为Sa，后手选择能从【i+1, j】得到的总量为$Sb$。那么$dp[i][j] = Sa - Sb$。下次选择相当于从【i+1, j】先手选择得到的总量为$Sb$，后手选择得到的总量为$Sa’$，其中$Sa - Pi = Sa'$：因为$Sa, Sa'$ 是同一个玩家的两次最优选择，第二次选择比第一次选择少了一个piles[i]。
根据dp的定义有：
$$
dp[i][j] = Sa - Sb \\
dp[i+1][j] = Sb - Sa' = Sb - Sa + piles[i] = -dp[i][j] + piles[i] \\
=> dp[i][j] = piles[i] - dp[i+1][j]
$$
对于先手选择 j, 推导过程相同，所以最后从两个的最大值中选择。
~~~C++
class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int len = piles.size();
        vector<vector<int>> dp(len, vector<int>(len));
        for(int i = 0; i < len; i++){
            dp[i][i] = piles[i];
        }

        for(int i = len - 2; i >= 0; i--){
            for(int j = i+1; j < len; j++){
                dp[i][j] = max(piles[i] - dp[i+1][j], piles[j] - dp[i][j-1]);
            }
        }
		// dp[0][len-1]表示从【0，len-1】的所有堆最佳选择的结果。
        return dp[0][len-1] > 0;
    }
};
~~~

#### 解法三：动态规划的行优化
~~~C++
class Solution {
public:
    bool stoneGame(vector<int>& piles) {
        int length = piles.size();
        auto dp = vector<int>(length);
        for (int i = 0; i < length; i++) {
            dp[i] = piles[i];
        }
        for (int i = length - 2; i >= 0; i--) {
            for (int j = i + 1; j < length; j++) {
                dp[j] = max(piles[i] - dp[j], piles[j] - dp[j - 1]);
            }
        }
        return dp[length - 1] > 0;
    }
};
~~~
