问题描述：给定n个括号，输出所有的可能的组合方式
解法1：
```
class Solution{
public:
	std::vector<std::string> generateParenthesis(int n){
		std::vector<std::vector<std::string> > dp(n + 1);
		dp[0].emplace_back("");
		for (int i = 1; i < n + 1; i++){
			for (int j = 0; j < i; j++){
				for (auto str_1 : dp[j]){
					for (auto str_2 : dp[i - 1 - j]){
						dp[i].emplace_back("(" + str_1 + ")" + str_2);
					}
				}
			}
		}
		return dp[n];
	}
};
```
