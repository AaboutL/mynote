# 1. 二叉树问题分类
二叉树问题归结为两类：
1. 通过遍历一遍二叉树得到结果，称为遍历
2. 通过子问题（子树）的结果推导出原问题的结果，称为分解

**遍历的问题**需要定义一个traverse()函数并配合一个外部变量，traverse()函数不需要返回值，直接修改外部变量
**分解的问题**定义一个递归函数，并充分利用这个函数的返回值。

# 2. 遍历位置不同产生的效果不同
```
void traverse(TreeNode root) {
    if (root == null) {
        return;
    }
    // 前序位置
    traverse(root.left);
    // 中序位置
    traverse(root.right);
    // 后序位置
}
```
## 2.1 前序位置
前序位置不会带来特殊的效果，前序位置的代码只能从函数参数中获得父节点传递来的数据。
## 2.2 中序位置
中序位置多用于BST（二叉搜索树），BST的中序遍历可以看做是遍历有序数组
## 2.3 后序位置
后序位置的代码不仅可以获取参数数据，还可以获取子树通过函数返回值传递回来的数据，这个返回值总结了我们需要的子树的信息。一旦发现题目和子树有关，那么大概率需要给函数设置合理的定义和返回值，在后序位置写代码。

## 示例
### 1. 【543. 二叉树的直径】https://leetcode.cn/problems/diameter-of-binary-tree/
~~~C++
class Solution {
public:
    int diameter=0;
    int diameterOfBinaryTree(TreeNode* root) {
        maxDepth(root);
        return diameter;
    }

    int maxDepth(TreeNode* root){
        if(!root)
            return 0;
        int leftDep = maxDepth(root->left);
        int rightDep = maxDepth(root->right);
        int curDep = max(leftDep, rightDep) + 1;
        diameter = max(diameter, leftDep + rightDep);
        return curDep;
    }
};
~~~

### 2. 【110. 平衡二叉树】https://leetcode.cn/problems/balanced-binary-tree/
~~~C++
class solution{
public:
	bool isBalanced(TreeNode* root){
		return traverse(root) == -1 ? false : true;
	}
	
	// traverse的返回值有两个含义：如果当前结点的左右子树都是平衡的，
	// 那么就返回当前结点的深度，否则返回-1，表示当前节点不平衡。
	int traverse(TreeNode* root){
		if(!root)
			return 0;
		int left = traverse(root->left);
		if(left == -1) return -1;
		int right = traverse(root->right);
		if(right == -1) return -1;
		return abs(right - left) < 2 ? max(right, left) + 1 : -1;
	}
}
~~~

### 3. 【116. 填充每个节点的下一个右侧节点指针】https://leetcode.cn/problems/populating-next-right-pointers-in-each-node/
#### 解法一：广度优先遍历：按层处理每个结点
~~~C++
class Solution {
public:
    Node* connect(Node* root) {
        if(!root)
            return root;
        int k = 1;
        int level = 0;
        queue<Node*> q;
        Node* cur = root;
        q.push(cur);
        while(!q.empty()){
            Node* cur = q.front();
            q.pop();
            k--;
            if(k!=0){
                cur->next = q.front();
            }
            else{
                cur->next = nullptr;
                level++;
                k = pow(2, level);
            }
            if(cur->left) q.push(cur->left);
            if(cur->right) q.push(cur->right);
        }
        return root;
    }
};
~~~
#### 解法二：深度优先遍历。
需要做一个转换：原始二叉树的深度优先遍历，无法处理相邻两个结点的子结点之间的关系。把相邻两个结点看做一个新结点，把这两个结点的4个子结点分成三个新子结点：（a.left, a.right）(a.right, b.left)  (b.left, b.right)。这样在遍历的时候，只需要把新结点中的两个结点连接起来
~~~C++
class Solution {
public:
  Node* connect(Node* root) {
	  traverse(root->left, root->right);
	  return root;
	}
	void traverse(Node* node1, Node* node2){
		if(!node1 && !node2)
			return;
		node1->next = node2;
		traverse(node1->left, node1->right);
		traverse(node1->right, node2->left);
		traverse(node2->left, node2->right);
	}
~~~
#### 解法三：
~~~C++
class Solution {
public:
    Node* connect(Node* root) {
        if(!root)
            return root;
        Node* cur = root;
        while(cur){
            for(Node* p = cur; p; p=p->next){
                if(p->left) p->left->next = p->right;
                if(p->right && p->next) p->right->next = p->next->left;
            }
            cur = cur->left;
        }
        return root;
    }
};
~~~
