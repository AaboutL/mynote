## 1\. 【36. 有效的数独】https://leetcode.cn/problems/valid-sudoku/

### 1.1 解法一：使用哈希表

```C++
bool isValidSudoku(vector<vector<char>>& board) {
    unordered_map<int, set<int>> row_map, col_map, box_map;
    for(int i = 0; i < 9; i++){
        row_map[i] = set<int>();
        col_map[i] = set<int>();
        box_map[i] = set<int>();
    }
    for(int i = 0; i < 9; i++){
        for(int j = 0; j < 9; j++){
            if(board[i][j] == '.') 
                continue;
            int num = board[i][j] - '0';
            int idx = i / 3 * 3 + j / 3;
            if(row_map[i].count(num) || col_map[j].count(num) || box_map[idx].count(num))
                return false;
            row_map[i].insert(num);
            col_map[j].insert(num);
            box_map[idx].insert(num);
        }
    }
    return true;
}
```

### 1.3 位运算
```C++
bool isValidSudoku(vector<vector<char>>& board) {
	// 每一行用一个二进制数表示，二进制数的某一位(idx)如果为1表示的意思
	// 是：1-9中的idx已经存在于这一行中。如：000100010：表示2和6已经
	// 在这一行中了，因为第2个位置和第6个位置为1。
	int row_map[9] = {0};
	int col_map[9] = {0};
	int box_map[9] = {0};
	for(int i = 0; i < 9; i++){
		for(int j = 0; j < 9; j++){
			if(board[i][j] == '.') 
				continue;
			int num = board[i][j] - '0';
			int idx = i / 3 * 3 + j / 3;
			if((row_map[i] >> num) & 1 == 1 || 
			  (col_map[j] >> num) & 1 == 1 ||
			  (box_map[idx] >> num) & 1 == 1)
				return false;
			row_map[i] |= (1 << num);
			col_map[j] |= (1 << num);
			box_map[idx] |= (1 << num);
		}
	}
	return true;
}
```
